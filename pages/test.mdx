# Button Component

Displays a button or a component that looks like a button.

## Text Button

import React from "react";
import primaryTextButtonMetaData from '../database/button/primaryTextButtonMetaData.ts';
import {CopyToClipboard} from 'react-copy-to-clipboard';
import ExampleComponent from '../components/shared/ExampleComponent.tsx'
import {PrimaryTextButtonLight, PrimaryTextButtonBlue, PrimaryTextButtonDark, 
      PrimaryRoundedTextButtonLight, PrimaryRoundedTextButtonBlue, PrimaryRoundedTextButtonDark, 
      SecondaryTextButtonLight, SecondaryTextButtonBlue, SecondaryTextButtonDark, 
      SecondaryRoundedTextButtonLight, SecondaryRoundedTextButtonBlue, SecondaryRoundedTextButtonDark, 
      GroupButtonLight, GroupButtonBlue, GroupButtonDark,
      PrimaryIconButtonLight, PrimaryIconButtonBlue, PrimaryIconButtonDark,
      PrimaryRoundedIconButtonLight, PrimaryRoundedIconButtonBlue, PrimaryRoundedIconButtonDark,
      SecondaryIconButtonLight,SecondaryIconButtonBlue,SecondaryIconButtonDark,
      SecondaryRoundedIconButtonLight,SecondaryRoundedIconButtonBlue,SecondaryRoundedIconButtonDark,
      PrimaryTextIconFrontButtonLight, PrimaryTextIconFrontButtonBlue, PrimaryTextIconFrontButtonDark,
      PrimaryRoundedTextIconFrontButtonLight, PrimaryRoundedTextIconFrontButtonBlue, PrimaryRoundedTextIconFrontButtonDark,
      SecondaryTextIconFrontButtonLight, SecondaryTextIconFrontButtonBlue, SecondaryTextIconFrontButtonDark,
      SecondaryRoundedTextIconFrontButtonLight, SecondaryRoundedTextIconFrontButtonBlue, SecondaryRoundedTextIconFrontButtonDark,
      PrimaryTextIconRearButtonLight, PrimaryTextIconRearButtonBlue, PrimaryTextIconRearButtonDark,
      PrimaryRoundedTextIconRearButtonLight, PrimaryRoundedTextIconRearButtonBlue, PrimaryRoundedTextIconRearButtonDark} from '../database/button/ButtonComponent.tsx'

export const PrimaryTextButton = () => {
  return (
    <>
    <div className='mb-20'>
        <ExampleComponent codeMetaData={primaryTextButtonMetaData} componentsArray={[<PrimaryTextButtonLight />, <PrimaryTextButtonBlue />, <PrimaryTextButtonDark />]}/>
    </div>
    </>
  );
};

<PrimaryTextButton/>
