# Button Component

Displays a button or a component that looks like a button.

## Primary Text Button

import React from "react";
import primaryTextButtonMetaData from '../database/button/primaryTextButtonMetaData.ts';
import {CopyToClipboard} from 'react-copy-to-clipboard';
import ExampleComponent from '../components/shared/ExampleComponent.tsx'
import {PrimaryTextButtonLight, PrimaryTextButtonBlue, PrimaryTextButtonDark, PrimaryRoundedTextButtonLight, PrimaryRoundedTextButtonBlue, PrimaryRoundedTextButtonDark, SecondaryTextButtonLight, SecondaryTextButtonBlue, SecondaryTextButtonDark, SecondaryRoundedTextButtonLight, SecondaryRoundedTextButtonBlue, SecondaryRoundedTextButtonDark} from '../database/button/ButtonComponent.tsx'

export const PrimaryTextButton = () => {
  return (
    <>
    <div className='mb-20'>
        <ExampleComponent codeMetaData={primaryTextButtonMetaData} componentsArray={[<PrimaryTextButtonLight />, <PrimaryTextButtonBlue />, <PrimaryTextButtonDark />]}/>
    </div>
    </>
  );
};

<PrimaryTextButton/>
